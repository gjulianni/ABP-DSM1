CREATE TABLE Usuário (
  idUsuário INTEGER NOT NULL,
  Nome VARCHAR(50) NULL,
  Email VARCHAR(50) NULL,
  DataCadastro DATE NULL,
  PRIMARY KEY(idUsuário)
);

CREATE TABLE Avaliação (
  idAvaliação INTEGER NOT NULL,
  idUsuário INTEGER NOT NULL,
  DataAvaliação DATE NULL,
  Score DECIMAL(5,2) NULL,
  PRIMARY KEY(idAvaliação),
  FOREIGN KEY(idUsuário) REFERENCES Usuário(idUsuário)
);

CREATE TABLE Questões (
  idQuestão INTEGER NOT NULL,
  idAvaliação INTEGER NOT NULL,
  Enunciado TEXT NULL,
  AlternativaCorreta CHAR(10) NULL,
  PRIMARY KEY(idQuestão),
  FOREIGN KEY(idAvaliação) REFERENCES Avaliação(idAvaliação)
);

CREATE TABLE Avaliação_Questões (
  idAvaliação INTEGER NOT NULL,
  idQuestão INTEGER NOT NULL,
  PRIMARY KEY(idAvaliação, idQuestão),
  FOREIGN KEY(idAvaliação) REFERENCES Avaliação(idAvaliação),
  FOREIGN KEY(idQuestão) REFERENCES Questões(idQuestão)
);
